```tsx
import React from "react";
import { motion } from "framer-motion";

interface InfoIconProps {
  size?: number;
  glow?: boolean;
  onClick?: () => void;
}

export const InfoIcon: React.FC<InfoIconProps> = ({
  size = 17,
  glow = true,
  onClick,
}) => {
  return (
    <motion.button
      onClick={onClick}
      whileTap={{ scale: 0.92 }}
      whileHover={{ scale: 1.08 }}
      className="relative flex items-center justify-center"
      style={{ width: size, height: size }}
    >
      {/* BACKGROUND CIRCLE */}
      <motion.div
        className="absolute inset-0 rounded-full bg-gray-400/20 border border-white/15 backdrop-blur-sm"
        animate={
          glow
            ? {
                boxShadow: [
                  "0 0 0px rgba(251,146,60,0.0)",
                  "0 0 8px rgba(251,146,60,0.45)",
                  "0 0 0px rgba(251,146,60,0.0)",
                ],
              }
            : {}
        }
        transition={{ duration: 2, repeat: Infinity, ease: "easeInOut" }}
      />

      {/* INFO CHARACTER */}
      <span
        className="relative z-10 text-[11px] font-bold text-gray-200"
        style={{ lineHeight: 1 }}
      >
        i
      </span>
    </motion.button>
  );
};
```

# ‚úÖ **FILE 2: InfoButton.tsx**

This wraps the icon, giving you a clean API:

**Location:**
`client/src/components/common/InfoButton.tsx`

```tsx
import React from "react";
import { InfoIcon } from "./InfoIcon";

interface InfoButtonProps {
  featureId: string; // e.g. "fridge-rescue"
  size?: number;
}

import { useCopilot } from "@/components/copilot/CopilotContext";

export const InfoButton: React.FC<InfoButtonProps> = ({
  featureId,
  size = 17,
}) => {
  const { runAction, open } = useCopilot();

  const handleClick = () => {
    open(); // open copilot sheet
    runAction({
      type: "run-command",
      id: `explain.${featureId}`,
    });
  };

  return <InfoIcon size={size} glow onClick={handleClick} />;
};
```

### This does EXACTLY what you want:

* Tap ‚Äúi‚Äù
  ‚Üí Copilot opens
  ‚Üí Copilot explains the feature
  ‚Üí Copilot walks the user through the screen
