Place in:
`client/src/components/ui/MedicalBadgeCard.tsx`

```tsx
import React from "react";

interface MedicalBadgeCardProps {
  title?: string;
  badges: { label: string; desc?: string; key?: string }[];
  className?: string;
  showGlow?: boolean; // auto-detect or override glow
}

const CRITICAL_BADGE_KEYS = [
  "cardiac",
  "heart",
  "heart-healthy",
  "renal",
  "kidney",
  "low sodium",
  "glp1",
  "glp-1",
  "diabetes",
  "diabetic",
];

export default function MedicalBadgeCard({
  title = "Medical Badges",
  badges,
  className = "",
  showGlow
}: MedicalBadgeCardProps) {

  // detect critical badges unless overridden
  const hasCritical =
    typeof showGlow === "boolean"
      ? showGlow
      : badges.some(b =>
          CRITICAL_BADGE_KEYS.some(k =>
            (b.key || b.label)?.toLowerCase().includes(k)
          )
        );

  return (
    <div
      className={`
        rounded-2xl p-4 max-w-xs text-white text-sm space-y-2
        border border-white/15
        bg-gradient-to-b from-white/5 via-black/80 to-black/95
        bg-clip-padding backdrop-blur-xl
        shadow-xl shadow-emerald-500/20
        ${hasCritical ? "medical-critical-card" : ""}
        ${className}
      `}
    >
      <div className="font-semibold text-white/90">{title}</div>

      {badges.length === 0 ? (
        <div className="text-white/60 text-xs">No health badges.</div>
      ) : (
        badges.map((item, i) => (
          <div key={item.key || i} className="flex items-start gap-2">
            <span className="inline-flex h-2 w-2 rounded-full bg-white/70 mt-1.5" />
            <div>
              <div className="font-medium">{item.label}</div>
              {item.desc && (
                <div className="text-white/80 text-xs">{item.desc}</div>
              )}
            </div>
          </div>
        ))
      )}
    </div>
  );
}
```
